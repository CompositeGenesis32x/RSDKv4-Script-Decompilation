// Aliases
private alias object.value0 : object.spinCount
private alias object.value1 : object.timer

private alias 0 : SIGNPOST_AWAITPLAYER
private alias 1 : SIGNPOST_SPINNING
private alias 2 : SIGNPOST_HANDLEACTFINISH
private alias 3 : SIGNPOST_EXITSTAGERIGHT
private alias 4 : SIGNPOST_TOSPECIALSTAGE

// Function declarations
reserve function SignPost_DebugDraw
reserve function SignPost_DebugSpawn

function SignPost_DebugDraw
	DrawSprite(0)
	DrawSprite(1)
end function


function SignPost_DebugSpawn
	CreateTempObject(TypeName[Sign Post], 0, object.xpos, object.ypos)
	object[tempObjectPos].frame = 1
	object[tempObjectPos].spinCount = 16
end function


event ObjectMain
	switch object.state
	case SIGNPOST_AWAITPLAYER
		temp0 = object.ypos
		temp0 += 0x400000
		if object[0].ypos < temp0
			stage.newXBoundary1 = object.xpos
			stage.newXBoundary1 >>= 16
			stage.newXBoundary1 -= screen.xcenter
			stage.newXBoundary1 -= screen.xcenter
			stage.newXBoundary2 = object.xpos
			stage.newXBoundary2 >>= 16
			stage.newXBoundary2 += screen.xcenter
			if object[0].xpos > object.xpos
				stage.newXBoundary1 = object.xpos
				stage.newXBoundary1 >>= 16
				stage.newXBoundary1 -= screen.xcenter
				
				if stage.debugMode == false
					if stage.listPos == 0
						if stage.minutes == 0
							if stage.seconds < 30
								CallNativeFunction2(SetAchievement, AchievementName[Blast Processing], 100)
							end if
						end if
					end if
				end if

				stage.timeEnabled = false
				object.state = SIGNPOST_SPINNING
				PlaySfx(SfxName[Sign Post], false)
				if PlayerObject_SuperState == SUPERSTATE_SUPER
					PlayerObject_SuperState = SUPERSTATE_FADEOUT
				end if
			end if
		end if
		break

	case SIGNPOST_SPINNING
		music.volume--
		object.timer++
		if object.timer == 2
			object.timer = 0
			object.frame++
			if object.frame == 5
				object.frame = 1
				Rand(temp0, 48)
				temp0 <<= 16
				temp0 += object.xpos
				temp0 -= 0x180000
				Rand(temp1, 32)
				temp1 <<= 16
				temp1 += object.ypos
				temp1 -= 0x140000
				CreateTempObject(TypeName[Ring Sparkle], 0, temp0, temp1)

				object.spinCount--
				if object.spinCount == 0
					if object[0].type == TypeName[Blank Object]
						object.state = SIGNPOST_TOSPECIALSTAGE

						if object.propertyValue < 3
							PlayMusic(TRACK_ACTFINISH)
						else
							if object.outOfBounds == false
								PlayMusic(TRACK_ACTFINISH)
							end if
						end if

						object[30].type = TypeName[Act Finish]
						object[30].drawOrder = 6
						object[30].priority = PRIORITY_ACTIVE
						object[0].controlMode = -1
						object[0].interaction = false
						if stage.player2Enabled == true
							object[1].interaction 	= false
							object[1].controlMode 	= -1
							object[1].up 			= false
							object[1].down 			= false
							object[1].left 			= false
							object[1].jumpPress 	= false
							object[1].jumpHold 		= false
						end if
					else
						object[0].controlMode 	= -1
						object[0].interaction 	= false
						object[0].up 			= false
						object[0].down 			= false
						object[0].left 			= false
						object[0].jumpPress 	= false
						object[0].jumpHold 		= false
						if stage.player2Enabled == true
							object[1].controlMode 	= -1
							object[1].interaction 	= false
							object[1].up 			= false
							object[1].down 			= false
							object[1].left 			= false
							object[1].jumpPress 	= false
							object[1].jumpHold 		= false
						end if

						object.state = SIGNPOST_HANDLEACTFINISH
					end if
				end if
			end if
		end if

		temp0 = 40
		temp0 += screen.xcenter
		temp0 <<= 16
		temp0 += object.xpos
		if object[0].xpos > temp0
			object[0].xpos = temp0
			object[0].speed = 0
			object[0].xvel = 0
		end if

		if stage.player2Enabled == true
			if object[1].xpos > temp0
				object[1].xpos = temp0
				object[1].speed = 0
				object[1].xvel = 0
			end if
		end if
		break

	case SIGNPOST_HANDLEACTFINISH
		if object[0].type == TypeName[Blank Object]
			object.state = SIGNPOST_TOSPECIALSTAGE
			if object.propertyValue < 3
				PlayMusic(TRACK_ACTFINISH)
			else
				if object.outOfBounds == false
					PlayMusic(TRACK_ACTFINISH)
				end if
			end if
			object[30].type = TypeName[Act Finish]
			object[30].drawOrder = 6
			object[30].priority = PRIORITY_ACTIVE
		end if

		if object[0].controlMode == -1
			object[0].right = true
		end if

		if stage.player2Enabled == true
			object[1].controlMode = object[0].controlMode
			if object[1].controlMode == -1
				if object[1].left == false
					object[1].right = true
				end if
			end if
		end if

		temp0 = screen.xcenter
		temp0 <<= 16
		temp0 += object.xpos
		if object[0].xpos > temp0
			if object.propertyValue < 3
				PlayMusic(TRACK_ACTFINISH)
			else
				if object.outOfBounds == false
					PlayMusic(TRACK_ACTFINISH)
				end if
			end if

			object.state = SIGNPOST_EXITSTAGERIGHT
			object[30].type = TypeName[Act Finish]
			object[30].drawOrder = 6
			object[30].priority = PRIORITY_ACTIVE
		end if
		break

	case SIGNPOST_EXITSTAGERIGHT
		if object[0].controlMode == -1
			object[0].right = true
		end if

		temp0 = 40
		temp0 += screen.xcenter
		temp0 <<= 16
		temp0 += object.xpos
		if object[0].xpos > temp0
			object[0].xpos = temp0
			object[0].speed = 0
			object[0].xvel = 0
		end if

		if stage.player2Enabled == true
			if object[1].controlMode == -1
				object[1].right = true
			end if

			if object[1].xpos > temp0
				object[1].xpos = temp0
				object[1].speed = 0
				object[1].xvel = 0
			end if
		end if
		break

	case SIGNPOST_TOSPECIALSTAGE
		if stage.player2Enabled == true
			temp0 = 40
			temp0 += screen.xcenter
			temp0 <<= 16
			temp0 += object.xpos
			object[1].right = true
			object[1].controlMode = -1
			if object[1].xpos > temp0
				object[1].xpos = temp0
				object[1].speed = 0
				object[1].xvel = 0
			end if
		end if
		break
	end switch
end event


event ObjectDraw
	DrawSprite(0)

	if object.spinCount > 8
		DrawSprite(object.frame)
	else
		temp0 = object.frame
		temp0 += 4
		DrawSprite(temp0)
	end if
end event

event RSDKEdit
	if editor.returnVariable == true
		switch editor.variableID
		case EDIT_VAR_PROPVAL // property value
			checkResult = object.propertyValue
			break
		case 0 // type
			checkResult = object.propertyValue
			break
		end switch
	else
		switch editor.variableID
		case EDIT_VAR_PROPVAL // property value
			object.propertyValue = editor.variableValue
			break
		case 0 // type
			object.propertyValue = editor.variableValue
			break
		end switch
	end if
end event
