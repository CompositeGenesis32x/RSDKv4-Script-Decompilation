event ObjectMain
	if object.propertyValue == 0
		object.xpos = player[-playerCount].xpos
		object.ypos = player[-playerCount].ypos
	end if

	switch object.state
	case 0
		if player[-playerCount].prevGravity == 0
			temp0 = player[-playerCount].speed
			Abs(temp0)
			if temp0 >= 0x80000
				object.drawOrder = DRAWORDER_PLAYER
				superSpark.timer = 0
				object.frame = 0
				object.state++
			end if
		end if
		break
	case 1
		superSpark.timer++
		if superSpark.timer >= 3
			superSpark.timer = 1
			object.frame++
			if object.frame >= 2
				object.state++
			end if
		end if
		break
	case 2
		superSpark.timer++
		if superSpark.timer >= 3
			superSpark.timer = 1
			object.frame--
			if object.frame <= 0
				object.state++
			end if
		end if
		break
	case 3
		superSpark.timer++
		if superSpark.timer >= 3
			superSpark.timer = 9
			object.state++
		end if
		break
	case 4
		if object.propertyValue == 1
			object.type = TypeName[Blank Object]
		end if

		superSpark.timer++
		if superSpark.timer >= 13
			temp0 = object[-playerCount].speed
			Abs(temp0)
			CheckLower(temp0, 0x80000)
			temp0 = checkResult
			CheckEqual(player[-playerCount].prevGravity, 0)
			temp0 &= checkResult
			if temp0 == 1
				object.state = 0
			else
				superSpark.timer = 0
				CreateTempObject(TypeName[Super Spark], 1, object[-playerCount].xpos, object[-playerCount].ypos)
				object[tempObjectPos].frame 	= 0
				object[tempObjectPos].state 	= 1
				object[tempObjectPos].drawOrder = player[-playerCount].sortedDrawOrder
			end if
		end if
		break
	end switch
end event


event ObjectDraw
	switch object.state
	case 1
	case 2
	case 3
		DrawSprite(object.frame)
		break
	end switch
end event
