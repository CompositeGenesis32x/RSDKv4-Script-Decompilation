private function Batbrain_DebugDraw
	DrawSpriteFX(0, FX_FLIP, object.xpos, object.ypos)
end function


private function Batbrain_DebugSpawn
	CreateTempObject(TypeName[Batbrain], 0, object.xpos, object.ypos)
	object[tempObjectPos].value0 = object.xpos
	object[tempObjectPos].value1 = object.ypos
end function


private function Batbrain_Function121
	temp1 = object.ixpos
	temp1 -= screen.xoffset
	CheckLower(temp1, 0)
	temp0 = checkResult
	
	CheckGreater(temp1, screen.xsize)
	temp0 |= checkResult
	
	temp1 = object.iypos
	temp1 -= screen.yoffset
	CheckLower(temp1, 0)
	temp0 |= checkResult
	
	CheckGreater(temp1, screen.ysize)
	checkResult |= temp0
end function

event ObjectUpdate
	switch object.state
	case 0
		object.priority = 1
		object.value4 = 0x7FFFFFFF
		object.value2 = 0
		object.value3 = 0xFFFF
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			temp0 = object[currentPlayer].xpos
			temp0 -= object.xpos
			Abs(temp0)
			if temp0 < object.value4
				object.value4 = temp0
				object.value2 = currentPlayer
			end if
			
			if temp0 < 0x800000
				if object[currentPlayer].ypos >= object.ypos
					if object.value3 != 0xFFFF
						arrayPos0 = object.value3
						if object[currentPlayer].ypos < object[arrayPos0].ypos
							object.value3 = currentPlayer
						end if
					else
						object.value3 = currentPlayer
					end if
				end if
			end if
		next
		
		if object.value3 != 0xFFFF
			arrayPos0 = object.value3
			temp0 = object[arrayPos0].ypos
			temp0 -= object.ypos
			if temp0 >= 0
				if temp0 <= 0x800000
					Rand(temp0, 8)
					if temp0 == 0
						object.state = 1
						object.value5 = object[arrayPos0].ypos
						object.animation = 1
					end if
				end if
			end if
		else
			arrayPos0 = object.value2
		end if
		
		if object[arrayPos0].xpos < object.xpos
			object.direction = 0
		else
			object.direction = 1
		end if
		break
		
	case 1
		object.ypos += object.yvel
		object.yvel += 0x1800
		arrayPos0 = object.value3
		if object[arrayPos0].xpos < object.xpos
			object.direction = 0
		else
			object.direction = 1
		end if
		temp0 = object.value5
		temp0 -= object.ypos
		if temp0 < 0x100000
			if object.direction == 0
				object.xvel = -0x10000
			else
				object.xvel = 0x10000
			end if
			object.yvel = 0
			object.animation = 2
			object.state = 2
		end if
		break
		
	case 2
		object.xpos += object.xvel
		arrayPos0 = object.value3
		temp0 = object[arrayPos0].xpos
		temp0 -= object.xpos
		Abs(temp0)
		if temp0 >= 0x800000
			Rand(temp0, 8)
			if temp0 == 0
				object.state = 3
			end if
		end if
		break
		
	case 3
		object.xpos += object.xvel
		object.ypos += object.yvel
		object.yvel -= 0x1800
		ObjectTileCollision(3, 0, -12, 0)
		if checkResult == 1
			object.xvel = 0
			object.yvel = 0
			object.state = 0
			object.animation = 0
		end if
		break
		
	case 4
		object.value0 = object.xpos
		object.value1 = object.ypos
		object.drawOrder = 5
		object.state = 0
		break
	end switch
	
	if object.outOfBounds == true
		temp0 = object.xpos
		temp1 = object.ypos
		object.xpos = object.value0
		object.ypos = object.value1
		if object.outOfBounds == true
			object.xvel = 0
			object.yvel = 0
			object.direction = 0
			object.animation = 0
			object.frame = 0
			object.priority = 0
			object.state = 0
		else
			object.xpos = temp0
			object.ypos = temp1
		end if
	end if
	
	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		BoxCollisionTest(C_TOUCH, object.entityPos, -12, -18, 12, 18, currentPlayer, object[currentPlayer].value40, object[currentPlayer].value38, object[currentPlayer].value41, object[currentPlayer].value39)
		if checkResult == true
			CreateTempObject(TypeName[Revive Enemy], object.propertyValue, object.value0, object.value1)
			object[tempObjectPos].priority = 1
			object[tempObjectPos].target = object.entityPos
			object[tempObjectPos].backupType = object.type
			object[tempObjectPos].backupState = 4
			CallFunction(Player_BadnikBreak)
		end if
	next
	
	switch object.animation
	case 0
		object.frame = 0
		break
		
	case 1
		object.frame = 1
		break
		
	case 2
		if object.animationTimer == 0
			CallFunction(Batbrain_Function121)
			if checkResult == false
				PlaySfx(SfxName[Batbrain Fly], 0)
			end if
		end if
		
		object.frame = object.animationTimer
		object.frame >>= 2
		object.frame++
		object.animationTimer++
		object.animationTimer &= 15
		break
		
	end switch
end event
