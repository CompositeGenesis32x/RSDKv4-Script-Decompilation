public function Motobug_DebugDraw
	DrawSpriteFX(0, FX_FLIP, object.xpos, object.ypos)
end function


public function Motobug_DebugSpawn
	CreateTempObject(TypeName[Motobug], 0, object.xpos, object.ypos)
	GetBit(temp0, object.direction, 0)
	if temp0 == 0
		object[tempObjectPos].direction = 0
		object[tempObjectPos].xvel = -0x10000
	else
		object[tempObjectPos].direction = 1
		object[tempObjectPos].xvel = 0x10000
	end if
end function


event ObjectUpdate
	switch object.state
	case 0
		object.priority = 1
		object.xpos += object.xvel
		ObjectTileGrip(0, 0, 15, 0)
		if checkResult == 0
			object.state = 1
			object.animation = 1
		end if
		if object.state == 1
			foreach (TypeName[MissionBlock], arrayPos0, ACTIVE_ENTITIES)
				temp1 = 0
				temp2 = object[arrayPos0].ixpos
				temp2 -= object.ixpos
				CheckGreater(temp2, -18)
				temp1 = checkResult
				CheckLower(temp2, 18)
				temp1 &= checkResult
				temp2 = object[arrayPos0].iypos
				temp2 -= object.iypos
				CheckGreater(temp2, 15)
				temp1 &= checkResult
				CheckLower(temp2, 40)
				temp1 &= checkResult
				if temp1 == 1
					object.state = 0
					object.animation = 0
				end if
			next
		end if
		if object.state == 0
			temp0 = 0
			foreach (TypeName[MissionBlock], arrayPos0, ACTIVE_ENTITIES)
				temp1 = 0
				temp2 = object[arrayPos0].ixpos
				temp2 -= object.ixpos
				CheckGreater(temp2, -36)
				temp1 = checkResult
				CheckLower(temp2, 36)
				temp1 &= checkResult
				temp2 = object[arrayPos0].iypos
				temp2 -= object.iypos
				CheckGreater(temp2, -30)
				temp1 &= checkResult
				CheckLower(temp2, 30)
				temp1 &= checkResult
				temp0 |= temp1
			next
			if temp0 != 0
				object.state = 1
				object.animation = 1
			end if
		end if
		if object.value3 < 20
			object.value3++
		else
			object.value3 = 0
			CreateTempObject(TypeName[Motobug Exhaust], 0, object.xpos, object.ypos)
			if object.direction == 0
				object[tempObjectPos].xpos += 0x160000
			else
				object[tempObjectPos].xpos -= 0x160000
			end if
			object[tempObjectPos].direction = object.direction
		end if
		break
	case 1
		if object.value0 < 59
			object.value0++
		else
			object.value0 = 0
			object.state = 0
			object.animation = 0
			object.direction ^= 1
			FlipSign(object.xvel)
		end if
		break
	end switch
	if object.outOfBounds == 1
		temp0 = object.xpos
		temp1 = object.ypos
		object.xpos = object.value1
		object.ypos = object.value2
		if object.outOfBounds == 1
			if object.propertyValue == 0
				object.direction = 0
				object.xvel = -0x10000
			else
				object.direction = 1
				object.xvel = 0x10000
			end if
			object.animation = 0
			object.priority = 0
			object.state = 0
		else
			object.xpos = temp0
			object.ypos = temp1
		end if
	end if
	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		BoxCollisionTest(C_TOUCH, object.entityPos, -14, -14, 14, 14, currentPlayer, object[currentPlayer].value40, object[currentPlayer].value38, object[currentPlayer].value41, object[currentPlayer].value39)
		if checkResult == 1
			CallFunction(PlayerObject_BadnikBreak)
		end if
	next
	if object.animation == 0
		object.frame = object.animationTimer
		object.frame >>= 3
		object.animationTimer++
		object.animationTimer &= 31
	end if
end event


event ObjectDraw
	if object.state != 2
		DrawSpriteFX(object.frame, FX_FLIP, object.xpos, object.ypos)
	end if
end event


event ObjectStartup
	if game.playMode == BOOT_PLAYMODE_MISSION
		LoadSpriteSheet("GHZ/Objects.gif")
		SpriteFrame(-21, -13, 40, 28, 98, 127)
		SpriteFrame(-20, -14, 39, 29, 139, 127)
		SpriteFrame(-20, -13, 39, 28, 98, 156)
		SpriteFrame(-20, -14, 39, 29, 139, 127)
	end if
	
	foreach (TypeName[Motobug], arrayPos0, ALL_ENTITIES)
		object[arrayPos0].value1 = object[arrayPos0].xpos
		object[arrayPos0].value2 = object[arrayPos0].ypos
		if object[arrayPos0].propertyValue == 0
			object[arrayPos0].direction = 0
			object[arrayPos0].xvel = -0x10000
		else
			object[arrayPos0].direction = 1
			object[arrayPos0].xvel = 0x10000
		end if
	next
	
	SetTableValue(TypeName[Motobug], DebugMode_ObjCount, DebugMode_TypesTable)
	SetTableValue(Motobug_DebugDraw, DebugMode_ObjCount, DebugMode_DrawTable)
	SetTableValue(Motobug_DebugSpawn, DebugMode_ObjCount, DebugMode_SpawnTable)
	DebugMode_ObjCount++
end event

event RSDKEdit
	if editor.returnVariable == true
		switch editor.variableID
		case EDIT_VAR_PROPVAL // property value
			checkResult = object.propertyValue
			checkResult &= 1
			break
		case 0 // direction
			checkResult = object.propertyValue
			checkResult &= 1
			break
		end switch
	else
		switch editor.variableID
		case EDIT_VAR_PROPVAL // property value
			object.propertyValue = editor.variableValue
			object.propertyValue &= 1
			break
		case 0 // direction
			object.propertyValue = editor.variableValue
			object.propertyValue &= 1
			break
		end switch
	end if
end event


event RSDKDraw
	GetBit(object.direction, object.propertyValue, 0)
	DrawSpriteFX(0, FX_FLIP, object.xpos, object.ypos)
end event

	AddEditorVariable("direction")

	AddEnumVariable("Left", 0)
	AddEnumVariable("Right", 1)
end event
