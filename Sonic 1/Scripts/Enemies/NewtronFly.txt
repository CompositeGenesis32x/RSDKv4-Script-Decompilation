function NewtronFly_DebugDraw
	DrawSprite(object.direction)
end function


function NewtronFly_DebugSpawn
	CreateTempObject(TypeName[Newtron Fly], object.direction, object.xpos, object.ypos)
	object[tempObjectPos].inkEffect = INK_ALPHA
	object[tempObjectPos].alpha = 0
	object[tempObjectPos].startPos.x = object.xpos
	object[tempObjectPos].startPos.y = object.ypos
end function


event ObjectMain
	switch object.state
	case NEWTRONFLY_WAITFORPLAYER
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_TOUCH, object.entityPos, -128, -64, 128, 64, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
			if checkResult == true
				object.state = NEWTRONFLY_APPEARING
			end if
		next
		break

	case NEWTRONFLY_APPEARING
		object.priority = PRIORITY_ACTIVE
		if object.alpha < 0xF8
			object.alpha += 4
			if object[0].xpos > object.xpos
				object.appearFrame = 0
			else
				object.appearFrame = 1
			end if
		else
			object.alpha = 0xFF
			object.animation = NEWTRONFLY_ANI_APPEAR
			if object.appearFrame == 0
				object.xvel = 0x20000
			else
				object.xvel = -0x20000
			end if
			object.direction = object.appearFrame
			object.state++
		end if
		break

	case NEWTRONFLY_APPEARED
		object.ypos += object.yvel
		object.yvel += 0x3800
		ObjectTileCollision(CSIDE_FLOOR, 0, 8, 0)
		if checkResult == true
			object.yvel = 0
			object.animationTimer = 0
			object.animation = NEWTRONFLY_ANI_MOVING
			object.state++
		end if
		break

	case NEWTRONFLY_MOVING
		object.xpos += object.xvel
		ObjectTileGrip(CSIDE_FLOOR, 0, 8, 0)
		break
	end switch

	if object.outOfBounds == true
		temp0 = object.xpos
		temp1 = object.ypos
		object.xpos = object.startPos.x
		object.ypos = object.startPos.y
		if object.outOfBounds == true
			object.yvel = 0
			object.xpos = object.startPos.x
			object.ypos = object.startPos.y
			object.animation = NEWTRONFLY_ANI_IDLE
			object.animationTimer = 0
			object.priority = PRIORITY_ACTIVE_BOUNDS
			object.alpha = 0
			object.state = NEWTRONFLY_WAITFORPLAYER
		else
			object.xpos = temp0
			object.ypos = temp1
		end if
	end if

	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		switch object.state
		default
		case NEWTRONFLY_WAITFORPLAYER
		case NEWTRONFLY_APPEARING
			break

		case NEWTRONFLY_APPEARED
		case NEWTRONFLY_MOVING
			BoxCollisionTest(C_TOUCH, object.entityPos, -16, -8, 16, 8, currentPlayer, player[currentPlayer].hitboxLeft, player[currentPlayer].hitboxTop, player[currentPlayer].hitboxRight, player[currentPlayer].hitboxBottom)
			if checkResult == true
				CallFunction(PlayerObject_BadnikBreak)
			end if
			break
		end switch
	next

	switch object.animation
	case NEWTRONFLY_ANI_IDLE
		break

	case NEWTRONFLY_ANI_APPEAR
		if object.animationTimer < 41
			object.frame = object.animationTimer
			object.frame /= 20
			object.animationTimer++
		end if
		object.frame += 2
		break

	case NEWTRONFLY_ANI_MOVING
		object.frame = object.animationTimer
		object.frame /= 3
		object.frame += 5
		object.animationTimer++
		object.animationTimer %= 6
		break
	end switch
end event


event ObjectDraw
	switch object.animation
	case NEWTRONFLY_ANI_IDLE
		DrawSpriteFX(object.appearFrame, FX_INK, object.xpos, object.ypos)
		break

	case NEWTRONFLY_ANI_APPEAR
		DrawSpriteFX(object.frame, FX_FLIP, object.xpos, object.ypos)
		break

	case NEWTRONFLY_ANI_MOVING
		DrawSpriteFX(object.frame, FX_FLIP, object.xpos, object.ypos)
		DrawSpriteFX(4, FX_FLIP, object.xpos, object.ypos)
		break
	end switch
end event
