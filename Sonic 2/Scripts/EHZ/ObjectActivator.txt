// this name isn't very descriptive, so while it "activates" objects, its usually used on both sides of a spiralPath obj to make sure its active

event ObjectMain
	arrayPos0 = object.propertyValue
	object[+arrayPos0].priority = PRIORITY_ACTIVE
end event

event RSDKDraw
	DrawSprite(0)
end event

event RSDKLoad
	LoadSpriteSheet("Global/Display.gif")
	SpriteFrame(-8, -8, 16, 16, 168, 18) // "Trigger" - #0
	
	SetVariableAlias(ALIAS_VAR_PROPVAL, "activateSlot")
end event
