private alias 51 : TYPE_WINDCURRENT

// ========================
// Function declarations

reserve function WindCurrent_Function120

// ========================
// Function Definitions
// ========================

public function WindCurrent_Function120
	object.gravity = 1
	if object.animation != ANI_CLINGING
		object.xvel = 0x80000
		object.speed = 0x80000
	else
		object.xvel = 0
		object.speed = 0
	end if
	if object.up == 1
		object.ypos -= 0x10000
	else
		if object.down == 1
			object.ypos += 0x10000
		end if
	end if
	object.yvel = 0
end function


// ========================
// Events
// ========================


event ObjectUpdate
	temp6 = 3
	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		CheckLower(object[currentPlayer].xpos, object.value0)
		temp0 = checkResult
		CheckGreater(object[currentPlayer].xpos, object.value1)
		temp0 |= checkResult
		CheckLower(object[currentPlayer].ypos, object.value2)
		temp0 |= checkResult
		CheckGreater(object[currentPlayer].ypos, object.value3)
		temp0 |= checkResult
		if temp0 == 0
			temp6 = 1
			CheckEqual(object[currentPlayer].state, Player_State_GotHit)
			temp0 = checkResult
			CheckEqual(object[currentPlayer].state, Player_State_Hurt)
			temp0 |= checkResult
			CheckEqual(object[currentPlayer].state, Player_State_Death)
			temp0 |= checkResult
			CheckEqual(object[currentPlayer].state, Player_State_Drown)
			temp0 |= checkResult
			CheckEqual(object[currentPlayer].state, WindCurrent_Function120)
			temp0 |= checkResult
			if temp0 == 0
				object[currentPlayer].state = WindCurrent_Function120
				object[currentPlayer].animation = ANI_FANROTATE
				object[currentPlayer].tileCollisions = 1
			end if
			if object[currentPlayer].animation == ANI_FANROTATE
				if object[currentPlayer].xvel == 0
					temp1 = object.value3
					temp1 -= object.value2
					temp1 >>= 1
					temp1 += object.value2
					if object[currentPlayer].ypos < temp1
						object[currentPlayer].ypos += 0x40000
					else
						object[currentPlayer].ypos -= 0x40000
					end if
				end if
			end if
		else
			if object[currentPlayer].state == WindCurrent_Function120
				
				// note to MegAmi - this is a state new to Origins, so i can't label it here
				// it's below Bored2
				object[currentPlayer].state = 12
				
				object[currentPlayer].xvel = 0x40000
				object[currentPlayer].speed = 0x40000
			end if
		end if
	next
	object.priority = temp6
	if object.propertyValue == 0
		object[+1].priority = temp6
	else
		object[-1].priority = temp6
	end if
end event


event ObjectStartup
	foreach (TypeName[Wind Current], arrayPos0, ALL_ENTITIES)
		if object[arrayPos0].propertyValue == 0
			arrayPos1 = arrayPos0
			arrayPos1++
			object[arrayPos0].value0 = object[arrayPos0].xpos
			object[arrayPos0].value1 = object[arrayPos1].xpos
			temp1 = object[arrayPos0].ypos
			temp2 = object[arrayPos1].ypos
		else
			arrayPos1 = arrayPos0
			arrayPos1--
			object[arrayPos0].value0 = object[arrayPos1].xpos
			object[arrayPos0].value1 = object[arrayPos0].xpos
			temp1 = object[arrayPos1].ypos
			temp2 = object[arrayPos0].ypos
		end if
		if temp1 < temp2
			object[arrayPos0].value2 = temp1
			object[arrayPos0].value3 = temp2
		else
			object[arrayPos0].value2 = temp2
			object[arrayPos0].value3 = temp1
		end if
		object[arrayPos0].priority = 3
	next
end event


// ========================

event RSDKEdit
	if editor.returnVariable == true
		switch editor.variableID
		case EDIT_VAR_PROPVAL // property value
			checkResult = object.propertyValue
			break
		case 0 // type
			checkResult = object.propertyValue
			break
		end switch
	else
		switch editor.variableID
		case EDIT_VAR_PROPVAL // property value
			object.propertyValue = editor.variableValue
			break
		case 0 // type
			object.propertyValue = editor.variableValue
			break
		end switch
	end if
end event


event RSDKDraw
	DrawSprite(0)
end event

	AddEditorVariable("type")
	SetActiveVariable("type")

	AddEnumVariable("Use Previous Object Slot", 1)
end event
