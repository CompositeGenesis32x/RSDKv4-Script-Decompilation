// States
private alias 0 : VFAN_DEACTIVATE
private alias 1 : VFAN_IDLE
private alias 2 : VFAN_ACTIVE

// Player Aliases
private alias object.value1 : player.timer

// Function declarations

	DrawSprite(0)
end function


function VFan_DebugSpawn
	CreateTempObject(TypeName[V Fan], 0, object.xpos, object.ypos)
	object[tempObjectPos].speed = 48
end function


event ObjectMain
	switch object.state
	case VFAN_DEACTIVATE
		if object.speed > 0
			object.speed--
		else
			object.state++
		end if
		object.frame = object.animationTimer
		object.frame /= 48
		object.animationTimer += object.speed
		object.animationTimer %= 288
		break

	case VFAN_IDLE
		object.timer++
		if object.timer == 30
			object.timer = 0
			object.speed = 48
			object.state++
		end if
		break

	case VFAN_ACTIVE
		object.frame = object.animationTimer
		object.frame /= 48
		object.animationTimer += object.speed
		object.animationTimer %= 288

		object.timer++
		if object.timer == 180
			object.timer = 0
			object.state = VFAN_DEACTIVATE
		end if

		temp0 = oscillation
		temp0 <<= 2
		GetTableValue(object.hitboxT, 10, StageSetup_oscillationTable)
		object.hitboxT >>= 8
		object.hitboxB = object.hitboxT
		object.hitboxT -= 112
		object.hitboxB += 32
		object.fanOffset.y = object.hitboxT
		object.fanOffset.y <<= 16
		object.fanOffset.y += object.ypos
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_TOUCH, object.entityPos, -64, object.hitboxT, 64, object.hitboxB, currentPlayer, 0, 0, 0, 0)
			if checkResult == true
				object[currentPlayer].animation 		= ANI_TWIRL
				object[currentPlayer].animationSpeed 	= 0x50
				object[currentPlayer].scrollTracking 	= true
				object[currentPlayer].gravity 			= GRAVITY_AIR
				object[currentPlayer].state 			= PlayerObject_HandleAir
				object[currentPlayer].tileCollisions 	= true
				object[currentPlayer].speed 			= object[currentPlayer].xvel
				player[currentPlayer].timer 			= 0
				object[currentPlayer].yvel 				= 0
				temp1 = object.fanOffset.y
				temp1 -= object[currentPlayer].ypos
				temp1 >>= 4
				object[currentPlayer].ypos += temp1
			end if
		next
		break
	end switch
end event

end event
