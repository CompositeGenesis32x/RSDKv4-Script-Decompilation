// Aliases
private alias object.value0 : object.offsetY

event ObjectMain
	if object.state == 0
		if object.offsetY > 0
			object.offsetY -= 0x80000
		end if
	else
		if object.offsetY < 0x400000
			object.offsetY += 0x80000
		else
			object.priority = PRIORITY_ACTIVE_BOUNDS
		end if
	end if

	object.ypos -= object.offsetY
	if object.offsetY == 0
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_BOX, object.entityPos, -8, -32, 8, 32, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
		next
	else
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_BOX, object.entityPos, -8, -32, 8, 8, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
		next
	end if
	object.ypos += object.offsetY
end event


event ObjectDraw
	temp0 = object.value0
	FlipSign(temp0)
	temp0 += object.ypos
	DrawSpriteFX(0, FX_FLIP, object.xpos, temp0)
end event


event ObjectStartup
	LoadSpriteSheet("MBZ/Objects.gif")

	SpriteFrame(-8, -32, 16, 64, 93, 312)
	foreach (TypeName[Secret Door], arrayPos0, ALL_ENTITIES)
		object[arrayPos0].drawOrder = 4
	next
end event

event RSDKDraw
	DrawSpriteFX(0, FX_FLIP, object.xpos, temp0)
end event

event RSDKLoad
	LoadSpriteSheet("MBZ/Objects.gif")

	SpriteFrame(-8, -32, 16, 64, 93, 312)
	
	SetVariableAlias(ALIAS_VAR_PROPVAL, "unused")
end event
